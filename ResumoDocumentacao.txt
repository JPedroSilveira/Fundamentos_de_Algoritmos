;; no-critério : (Número Número -> Booleano) Lista-de-números Número -> Lista-de-números
;; A partir da lista dada, gera uma lista com os valores
;; dentro de um critério em relação a um dado valor
(define (no-critério op ldn valor)
	(cond
		[(empty? ldn) empty]
		[(op (first ldn) valor)
			(cons (first ldn) (no-critério op (rest ldn) valor))]
		[else (no-critério op (rest ldn) valor)]))
